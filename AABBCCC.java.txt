package practice;

public class AABBBCC {

	public static void main(String[] args) {
		String st="AABBBCCCDDEEE";
		String s1="";
		for (int i = 0; i < st.length(); i++) 
		{
			s1+=st.charAt(i);
			int count=0;
			for (int j = i; j < st.length(); j++)
			{
				if(st.charAt(i)==st.charAt(j))
				{
					i=j;
					count++;
				}
			}
			
			s1+=count;
		}
		System.out.println(s1);
	}

}
